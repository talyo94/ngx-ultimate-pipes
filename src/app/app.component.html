Examples:

EmptyPipe ('empty')

[1, 2, 3] -> {{ [1, 2, 3] | empty }} // false

[] -> {{ [] | empty }} // true

{{ { objects: 'works'} | empty }} // false

{{ {} | empty }} // true

FilterPipe ('filter')

JoinPipe ('join')

[1, 2, 3] -> {{ [1, 2, 3] | join: '@' }} // '1@2@3'

['first', 'second', 'third'] -> {{ ['first', 'second', 'third'] | join: '' }} // 'firstsecondthird'

ReverseElementsPipe ('reverseelements')

[1, 2, 3] ->{{ [1, 2, 3] | reverseelements }} // [3, 2, 1]

tally ('tally')

['apple', 'banana', 'orange', 'tomato', 'apple', 'apple', 'apple', 'banana', 'orange', 'banana', 'orange', 'cucumber'] -> {{['apple', 'banana', 'orange', 'tomato', 'apple', 'apple', 'apple', 'banana', 'orange', 'banana', 'orange', 'cucumber'] | tally | json}}

YesNoPipe ('yesno')

true | yesno -> {{ true | yesno }}

false | yesno -> {{ false | yesno }}

true | yesno: ['yeah', 'nope'] -> {{ true | yesno: ['yeah', 'nope'] }}

false | yesno: ['yeah', 'nope'] -> {{ false | yesno: ['yeah', 'nope'] }}

undefied | yesno -> {{ undefied | yesno }}

undified | yesno: ['yeah', 'nope', 'maybe'] -> {{ undified | yesno: ['yeah', 'nope', 'maybe'] }}
